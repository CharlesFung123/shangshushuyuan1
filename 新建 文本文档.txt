    <script>
        // 管理员默认密码（可自行修改）
        const ADMIN_PASSWORD = "admin123";
        let currentUploadObj = null;
        let currentUploadType = "";

        // 生成100个学员位置 + 500个碑帖位置
        function makeBox(el,num){
            let h = '';
            for(let i=0;i<num;i++) h+=`<img class="grid-item" onclick="openAdminUpload(this)">`;
            document.getElementById(el).innerHTML = h;
        }
        makeBox('sbox',100);
        makeBox('bbox',500);

        // 页面切换
        function show(id){
            document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            window.scrollTo(0,0);
        }

        // 打开管理员密码验证弹窗（单张上传）
        function openAdminUpload(obj) {
            currentUploadObj = obj;
            document.getElementById('passwordModal').style.display = 'flex';
        }

        // 打开管理员密码验证弹窗（批量上传）
        function openAdminBatchUpload(type = "") {
            currentUploadType = type;
            document.getElementById('passwordModal').style.display = 'flex';
        }

        // 验证管理员密码
        function checkPassword() {
            const inputPwd = document.getElementById('adminPwd').value;
            if (inputPwd === ADMIN_PASSWORD) {
                // 密码正确，关闭弹窗并执行上传
                document.getElementById('passwordModal').style.display = 'none';
                document.getElementById('adminPwd').value = '';
                
                if (currentUploadObj) {
                    // 单张上传
                    const input = document.createElement('input');
                    input.type = 'file';
                    input.accept = 'image/*';
                    input.onchange = (e) => {
                        const reader = new FileReader();
                        reader.onload = () => {
                            currentUploadObj.src = reader.result;
                            currentUploadObj = null;
                        };
                        reader.readAsDataURL(e.target.files[0]);
                    };
                    input.click();
                } else if (currentUploadType) {
                    // 批量上传
                    let fileInputId = '';
                    if (currentUploadType === 'stu') {
                        fileInputId = 'file-stu';
                    } else if (currentUploadType === 'bei') {
                        fileInputId = 'file-bei';
                    } else {
                        fileInputId = 'file-teacher';
                    }
                    document.getElementById(fileInputId).click();
                    currentUploadType = "";
                }
            } else {
                alert('密码错误！请输入正确的管理员密码');
                document.getElementById('adminPwd').value = '';
            }
        }

        // 批量上传图片
        function setImg(cls,inp){
            const files = inp.files;
            const imgs = document.querySelectorAll(cls);
            for(let i=0;i<files.length;i++){
                if(i>=imgs.length)break;
                const reader = new FileReader();
                reader.onload = (e) => imgs[i].src = e.target.result;
                reader.readAsDataURL(files[i]);
            }
            inp.value = '';
        }

        // 提交报名表单
        function submitSignup() {
            const name = document.getElementById('name').value;
            const phone = document.getElementById('phone').value;
            const course = document.getElementById('course').value;
            const age = document.getElementById('age').value;

            if (!name || !phone || !course || !age) {
                alert('请填写完整的报名信息！');
                return;
            }

            // 模拟提交成功
            alert(`报名成功！\n姓名：${name}\n电话：${phone}\n课程：${course}\n年龄：${age}\n我们将尽快与您联系！`);
            
            // 重置表单
            document.getElementById('name').value = '';
            document.getElementById('phone').value = '';
            document.getElementById('course').value = '';
            document.getElementById('age').value = '';
            document.getElementById('remark').value = '';
        }
    </script>
